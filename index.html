# pretty-picture.py
import turtle

tess = turtle.Turtle()
wn = turtle.Screen()

tess.speed(10)
tess.pensize(2)
tess.color('blue')
wn.bgcolor('lightgreen')

def drawSquare(t: turtle.Turtle, sz):
    for i in range(4):
        t.fd(sz)
        t.left(90)
        t.bk(sz*.5)

def drawGrid(t, sz):
    for i in range(4):
        drawSquare(t, sz)
        t.left(60)

def main():
    for i in range(50):
        drawGrid(tess, 100)
        tess.left(6)


tess.speed(0)
main()
